Koneksi Spreadsheet

Hubungkan sistem ini secara langsung ke Google Sheets. Semua transaksi yang diinput di dashboard akan langsung tersinkronisasi secara cloud.

Google Apps Script (Salin ke Editor Apps Script Anda)

// --- SISTEM KEUANGAN RIZKAN (BACKEND SPREADSHEET SYNC) ---
// 1. Buat Google Spreadsheet baru.
// 2. Klik Ekstensi -> Apps Script.
// 3. Hapus kode bawaan, tempel kode ini.
// 4. Klik "Terapkan" -> "Penerapan Baru" -> Pilih jenis "Aplikasi Web".
// 5. Ubah akses ke "Siapa saja", lalu klik Terapkan & berikan izin.
// 6. Salin URL Aplikasi Web yang dihasilkan dan masukkan ke setelan tab sebelah kiri.

function doGet(e) {
  var sheet = getOrCreateSheet();
  var rows = sheet.getDataRange().getValues();
  var data = [];
  
  // Lewati baris header pertama
  for (var i = 1; i < rows.length; i++) {
    data.push({
      id: rows[i][0],
      tanggal: rows[i][1],
      tipe: rows[i][2],
      kategori: rows[i][3],
      jumlah: Number(rows[i][4]),
      catatan: rows[i][5]
    });
  }
  
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var sheet = getOrCreateSheet();
    
    if (params.action === "clear") {
      // Clear all rows except header
      var lastRow = sheet.getLastRow();
      if (lastRow > 1) {
        sheet.deleteRows(2, lastRow - 1);
      }
      return ContentService.createTextOutput(JSON.stringify({status: "success", message: "Cleared all database records"}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    // Simpan Baris Baru
    sheet.appendRow([
      params.id || new Date().getTime(),
      params.tanggal || new Date().toLocaleString(),
      params.tipe,
      params.kategori,
      params.jumlah,
      params.catatan
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({status: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function getOrCreateSheet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Transaksi_Rizkan");
  if (!sheet) {
    sheet = ss.insertSheet("Transaksi_Rizkan");
    sheet.appendRow(["ID", "Tanggal", "Tipe", "Kategori", "Jumlah", "Catatan"]);
  }
  return sheet;
}
