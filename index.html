Instalasi Integrasi WhatsApp & Google Sheet Gratis

Kami menggunakan skema **Twilio Developer Sandbox** (Gratis & Bebas Banned Permanent) yang terhubung ke **Google Apps Script** untuk menulis ke Google Spreadsheet secara instan tanpa perlu membiarkan ponsel Anda menyala/terkoneksi!

1

Siapkan Google Sheet

Buat Google Spreadsheet baru. Atur nama lembar kerja (sheet) menjadi "Transaksi".

2

Pasang Apps Script

Pilih menu **Ekstensi > Apps Script**. Salin & tempel kode di bawah ini, lalu klik **Terapkan sebagai Web App** (Setel akses ke 'Anyone').

3

Sambungkan ke Twilio

Buat akun Twilio gratis. Buka WhatsApp Sandbox, lalu masukkan URL Web App Google Apps Script ke kolom *Webhook URL* di Twilio.

Kode Google Apps Script (code.gs) 
/* Google Apps Script untuk Integrasi WhatsApp Bot Keuangan Rizkan (Dynamic Category & Sign Support) */

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Transaksi");
  if (!sheet) {
    sheet = ss.insertSheet("Transaksi");
    sheet.appendRow(["Tanggal", "Tipe", "Kategori", "Jumlah", "Deskripsi", "Sumber"]);
  }
  
  // Ambil atau buat lembar master Kategori secara dinamis
  var catSheet = ss.getSheetByName("Kategori");
  if (!catSheet) {
    catSheet = ss.insertSheet("Kategori");
    catSheet.appendRow(["Kategori", "Batas Anggaran"]);
    catSheet.appendRow(["Makanan", 2500000]);
    catSheet.appendRow(["Transport", 1000000]);
    catSheet.appendRow(["Tagihan", 2000000]);
    catSheet.appendRow(["Hiburan", 2500000]);
  }
  
  // Ambil parameter dari pesan WhatsApp masuk (Twilio)
  var messageBody = e.parameter.Body.trim();
  var sender = e.parameter.From;
  var responseText = "";
  
  try {
    var cleanMsg = messageBody.toLowerCase();
    var parts = messageBody.split(/\s+/);
    var firstPart = parts[0].toLowerCase();
    
    // Deteksi jika input menggunakan shortcut cepat seperti -50000 atau +100000
    var isShorthand = /^[+-]\d+/.test(parts[0]);
    
    if (cleanMsg === "/rekap" || cleanMsg === "rekap") {
      responseText = generateRekap(sheet);
    } else if (cleanMsg === "/alokasi" || cleanMsg === "alokasi") {
      responseText = generateAlokasi(sheet, catSheet);
    } else if (firstPart === "/tambahkategori" || firstPart === "tambahkategori") {
      responseText = addNewCategory(messageBody, catSheet);
    } else if (firstPart === "/pemasukan" || firstPart === "pemasukan" || (isShorthand && parts[0].indexOf("+") === 0)) {
      responseText = parseTransaction(messageBody, "Pemasukan", sheet, catSheet, isShorthand);
    } else if (firstPart === "/pengeluaran" || firstPart === "pengeluaran" || (isShorthand && parts[0].indexOf("-") === 0)) {
      responseText = parseTransaction(messageBody, "Pengeluaran", sheet, catSheet, isShorthand);
    } else {
      responseText = "💡 *Keuangan Rizkan Bot* \n\n" +
                     "Format Perintah (Bisa Tanpa '/' atau dengan angka minus/plus):\n" +
                     "➖ `pengeluaran -50000 Makanan beli bakso`\n" +
                     "➕ `pemasukan 500000 Gaji bonus bulanan`\n" +
                     "📁 `tambahkategori [nama_kategori] [limit]`\n\n" +
                     "⚡ *Shortcut Super Cepat:* \n" +
                     "👉 `-50000 Tagihan bayar wifi` (Pengeluaran)\n" +
                     "👉 `+100000 Gaji saku tambahan` (Pemasukan)\n\n" +
                     "📋 `rekap` - Lihat saldo dan sisa budget\n" +
                     "📊 `alokasi` - Lihat kondisi anggaran";
    }
  } catch(err) {
    responseText = "❌ Gagal memproses perintah. Pastikan format penulisan benar.";
  }
  
  // Format balasan XML untuk Twilio
  var xmlResponse = "<Response><Message>" + responseText + "</Message></Response>";
  return ContentService.createTextOutput(xmlResponse).setMimeType(ContentService.MimeType.XML);
}

function addNewCategory(text, catSheet) {
  var parts = text.split(/\s+/);
  if (parts.length < 3) return "❌ Format salah. Contoh: `tambahkategori Kesehatan 1500000`";
  
  var name = parts[1];
  var limit = parseInt(parts[2].replace(/[^0-9]/g, ""));
  
  // Cari apakah kategori sudah ada
  var data = catSheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0].toLowerCase() === name.toLowerCase()) {
      catSheet.getRange(i + 1, 2).setValue(limit);
      return "✅ *Kategori Anggaran '" + name + "' Berhasil Diupdate!* \nLimit bulanan baru: Rp " + limit.toLocaleString("id-ID");
    }
  }
  
  catSheet.appendRow([name, limit]);
  return "📁 *Kategori Anggaran '" + name + "' Berhasil Ditambahkan!* \nLimit bulanan: Rp " + limit.toLocaleString("id-ID");
}

function parseTransaction(text, type, sheet, catSheet, isShorthand) {
  var parts = text.split(/\s+/);
  var amount = 0;
  var categoryInput = "";
  var description = "";
  
  if (isShorthand) {
    amount = Math.abs(parseInt(parts[0].replace(/[^0-9]/g, "")));
    categoryInput = parts[1] || "Lainnya";
    description = parts.slice(2).join(" ") || "-";
  } else {
    if (parts.length < 3) return "❌ Format salah. Contoh: `pengeluaran -50000 Tagihan bayar air` atau `-50000 Tagihan bayar air`";
    amount = Math.abs(parseInt(parts[1].replace(/[^0-9]/g, "")));
    categoryInput = parts[2];
    description = parts.slice(3).join(" ") || "-";
  }
  
  var formattedDate = Utilities.formatDate(new Date(), "GMT+7", "yyyy-MM-dd HH:mm:ss");
  
  // Cari atau buat kategori otomatis di database master jika belum terdaftar
  var data = catSheet.getDataRange().getValues();
  var matchedCategory = "";
  for (var i = 1; i < data.length; i++) {
    if (data[i][0].toLowerCase() === categoryInput.toLowerCase() || data[i][0].toLowerCase().indexOf(categoryInput.toLowerCase()) !== -1) {
      matchedCategory = data[i][0];
      break;
    }
  }
  
  if (matchedCategory === "") {
    matchedCategory = categoryInput.charAt(0).toUpperCase() + categoryInput.slice(1).toLowerCase();
    catSheet.appendRow([matchedCategory, 1000000]); // default limit 1 juta
  }
  
  // Tambah baris ke Google Sheets
  sheet.appendRow([formattedDate, type, matchedCategory, amount, description, "WhatsApp Bot"]);
  
  return "💰 *" + type + " Berhasil Dicatat!* \n\n" +
         "📅 Tanggal: " + formattedDate + "\n" +
         "🏷️ Kategori: " + matchedCategory + "\n" +
         "💵 Jumlah: Rp " + amount.toLocaleString("id-ID") + "\n" +
         "📝 Ket: " + description + "\n\n" +
         "Data otomatis sinkron ke Google Sheet!";
}

function generateAlokasi(sheet, catSheet) {
  var txData = sheet.getDataRange().getValues();
  var catData = catSheet.getDataRange().getValues();
  
  var budgets = {};
  for (var i = 1; i < catData.length; i++) {
    budgets[catData[i][0]] = { limit: parseInt(catData[i][1]), spent: 0 };
  }
  
  for (var j = 1; j < txData.length; j++) {
    var type = txData[j][1];
    var category = txData[j][2];
    var amount = parseInt(txData[j][3]);
    
    if (type === "Pengeluaran") {
      if (budgets[category] !== undefined) {
        budgets[category].spent += amount;
      }
    }
  }
  
  var response = "📊 *KONDISI ALOKASI ANGGARAN:* \n\n";
  for (var cat in budgets) {
    var item = budgets[cat];
    var pct = item.limit > 0 ? ((item.spent / item.limit) * 100).toFixed(0) : 0;
    var warn = pct >= 100 ? "⚠️ (HABIS!)" : "";
    response += "🏷️ *" + cat + ":* Rp " + item.spent.toLocaleString("id-ID") + " / Rp " + item.limit.toLocaleString("id-ID") + " (*" + pct + "%*) " + warn + "\n";
  }
  return response;
}

function generateRekap(sheet) {
  var data = sheet.getDataRange().getValues();
  var totalIncome = 0;
  var totalExpense = 0;
  
  for (var i = 1; i < data.length; i++) {
    var type = data[i][1];
    var amount = parseInt(data[i][3]);
    if (type === "Pemasukan") totalIncome += amount;
    else if (type === "Pengeluaran") totalExpense += amount;
  }
  
  var balance = totalIncome - totalExpense;
  return "📋 *REKAP KEUANGAN KEUANGAN RIZKAN* \n\n" +
         "🟢 Total Pemasukan: Rp " + totalIncome.toLocaleString("id-ID") + "\n" +
         "🔴 Total Pengeluaran: Rp " + totalExpense.toLocaleString("id-ID") + "\n" +
         "━━━━━━━━━━━━━━━━━━\n" +
         "💰 *SISA SALDO: Rp " + balance.toLocaleString("id-ID") + "*\n\n" +
         "Tetap hemat & bijak menggunakan uang!";
}
